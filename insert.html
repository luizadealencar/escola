Exemplo de Inserção de Dados na Tabela em SQL!

Funcionario


insert into Funcionario (codigo_func, nome_func) values (1,"João"), (2,"Maria");

Funcionario
codigo_func nome_func
01 João
02 Maria


Projeto


insert into Projeto (codigo_proj, nome_proj) values (100,"viva"), (200,"natureza");

Projeto
codigo_prof nome_proj
100 viva
200 natureza


É Alocado


insert into Alocado (codigo_func, codigo_proj) values (01, 100), (02, 200);

É Alocado
codigo_func codigo_proj
01 100
02 200
