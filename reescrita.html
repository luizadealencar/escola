Exemplo de Método de Reescrita em Java!

Arquivo: Pessoa.java


public class Pessoa {
    String nome;
    String cpf;
    char sexo;
    int idade;

    public Pessoa(String nome, String cpf, char sexo, int idade) {
        this.nome = nome;
        this.cpf = cpf;
        this.sexo = sexo;
        this.idade = idade;
    }

    public void aniversario() {
        this.idade++;
        System.out.println(nome + " fez aniversário e agora tem " + idade + " anos.");
    }

    public void alteraCPF(String cpf) {
        this.cpf = cpf;
        System.out.println("CPF de " + nome + " foi alterado para: " + this.cpf);
    }

    public int consultaIdade() {
        return this.idade;
    }

    // Getters e setters
    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public String getCpf() {
        return cpf;
    }

    public void setCpf(String cpf) {
        this.cpf = cpf;
    }

    public char getSexo() {
        return sexo;
    }

    public void setSexo(char sexo) {
        this.sexo = sexo;
    }

    public int getIdade() {
        return idade;
    }

    public void setIdade(int idade) {
        this.idade = idade;
    }
}

Arquivo: Aluno.java


public class Aluno extends Pessoa {
    String matricula;

    public Aluno(String nome, String cpf, char sexo, int idade, String matricula) {
        super(nome, cpf, sexo, idade);
        this.matricula = matricula;
    }

    @Override
    public void aniversario() {
        super.aniversario();
        System.out.println("O aluno " + getNome() + " tem " + getIdade() + " anos agora.");
    }

    @Override
    public void alteraCPF(String cpf) {
        super.alteraCPF(cpf);
        System.out.println("O CPF do aluno " + getNome() + " foi atualizado para: " + getCpf());
    }

    // Getter e setter
    public String getMatricula() {
        return matricula;
    }

    public void setMatricula(String matricula) {
        this.matricula = matricula;
    }
}

Arquivo: Professor.java


public class Professor extends Pessoa {
    double salario;

    public Professor(String nome, String cpf, char sexo, int idade, double salario) {
        super(nome, cpf, sexo, idade);
        this.salario = salario;
    }

    @Override
    public void aniversario() {
        super.aniversario();
        System.out.println("O professor " + getNome() + " agora tem " + getIdade() + " anos.");
    }

    @Override
    public void alteraCPF(String cpf) {
        super.alteraCPF(cpf);
        System.out.println("O CPF do professor " + getNome() + " foi atualizado para: " + getCpf());
    }

    // Getter e setter
    public double getSalario() {
        return salario;
    }

    public void setSalario(double salario) {
        this.salario = salario;
    }
}

Arquivo Main.java


public class Main {
    public static void main(String[] args) {
        Aluno aluno = new Aluno("Carlos", "123.456.789-00", 'M', 20, "20210001");
        Professor professor = new Professor("Mariana", "987.654.321-00", 'F', 35, 5000.0);

        System.out.println("### Aniversário do Aluno ###");
        aluno.aniversario();

        System.out.println("\n### Aniversário do Professor ###");
        professor.aniversario();

        System.out.println("\n### Alteração de CPF do Aluno ###");
        aluno.alteraCPF("111.222.333-44");

        System.out.println("\n### Alteração de CPF do Professor ###");
        professor.alteraCPF("555.666.777-88");
    }
}
